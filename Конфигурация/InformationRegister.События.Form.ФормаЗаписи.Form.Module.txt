
&НаСервере
Процедура ПриСозданииНаСервере(Отказ, СтандартнаяОбработка)
	//Вставить содержимое обработчика
	Если Параметры.Свойство("ДатаЗаявки") тогда
		ЭтаФорма.Запись.ДатаЗаявки = Параметры.ДатаЗаявки;
		ЭтаФорма.Запись.Период = ТекущаяДата();
	КонецЕсли;
	
	Если Параметры.Свойство("НомерЗаявки") тогда
		ЭтаФорма.Запись.НомерЗаявки = Параметры.НомерЗаявки;
	КонецЕсли;
	
	Если Параметры.Свойство("Контрагент") тогда
		ЭтаФорма.Запись.Контрагент = Параметры.Контрагент;
	КонецЕсли;
	
КонецПроцедуры


&НаКлиенте
Процедура ВыездВнедренцаПриИзменении(Элемент)
	ЭтаФорма.Элементы.стрДоговоренности.Видимость = ЭтаФорма.Запись.ВыездВнедренца;
	ЭтаФорма.Элементы.стрЗадачиОП.Видимость = ЭтаФорма.Запись.ВыездВнедренца;
	
КонецПроцедуры


&НаКлиенте
Процедура ПриОткрытии(Отказ)
	ЭтаФорма.Элементы.стрДоговоренности.Видимость = ЭтаФорма.Запись.ВыездВнедренца;
	ЭтаФорма.Элементы.стрЗадачиОП.Видимость = ЭтаФорма.Запись.ВыездВнедренца;
КонецПроцедуры
